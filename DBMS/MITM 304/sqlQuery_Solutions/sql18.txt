DECLARE
 CURSOR C_ZIP IS
 SELECT * FROM EMPLOYEES;
 VR_ZIP C_ZIP%ROWTYPE;
 --EMPLOYEES%ROWTYPE;
 
 BEGIN	
	OPEN C_ZIP;
	LOOP
		FETCH C_ZIP INTO VR_ZIP;
		EXIT WHEN C_ZIP%NOTFOUND;
		DBMS_OUTPUT.PUT_LINE(VR_ZIP.LAST_NAME ||' '|| VR_ZIP.SALARY ||' '|| (VR_ZIP.SALARY+500));

	END LOOP;
	
	
END;
