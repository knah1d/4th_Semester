CREATE OR REPLACE PROCEDURE INSRT_MYTAB
(P_ID IN MYTAB.ID%TYPE,
P_NAME IN MYTAB.NAME%TYPE,
P_LOC IN MYTAB.LOC%TYPE )
IS 
BEGIN
	INSERT INTO MYTAB 
	(ID,NAME,LOC)
	VALUES(P_ID,P_NAME,P_LOC);
END INSRT_MYTAB;
/
DECLARE
 P_ID NUMBER :=5;
 P_NAME VARCHAR2(25):='ABC';
 P_LOC VARCHAR2(50):='DEF';
 
 BEGIN	
	INSRT_MYTAB(P_ID,P_NAME,P_LOC);
	DBMS_OUTPUT.PUT_LINE('SUCCESSFULLY INSERT ');
END;
/
