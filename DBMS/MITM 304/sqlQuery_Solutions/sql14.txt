CREATE OR REPLACE FUNCTION GET_AREA
(P_RADIUS NUMBER ) RETURN NUMBER IS
V_AREA  NUMBER:=0;
BEGIN
	V_AREA:=(22/7)*P_RADIUS*P_RADIUS;
	RETURN V_AREA;
END GET_AREA;
/
DECLARE 
P_RADIUS NUMBER:=3;
V_AREA  NUMBER;
BEGIN	
	V_AREA:=GET_AREA(P_RADIUS);
	DBMS_OUTPUT.PUT_LINE('AREA OF THE CIRCLE IS '||V_AREA);
END;
/
